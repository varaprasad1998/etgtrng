1.SELECT w.WORKER_ID, w.FIRST_NAME+' '+LAST_NAME as 'Fullname',w.DEPARTMENT,t.WORKER_TITLE
FROM WORKER_DETAILS w,TITLE t
WHERE w.WORKER_ID = t.WORKER_REF_ID AND WORKER_TITLE ='Consultant';



2.WITH HIGHEST AS (
    SELECT DEPARTMENT,
	FIRST_NAME , 
	SALARY,
    DENSE_RANK() OVER(PARTITION BY DEPARTMENT ORDER BY SALARY DESC) AS RowNum
        FROM Worker_Details
)
SELECT DEPARTMENT, FIRST_NAME, SALARY
    FROM HIGHEST
    WHERE RowNum = 1;
